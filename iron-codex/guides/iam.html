Implement Multi-Factor Authentication (MFA)

Critical Medium Difficulty

Requirement: Enforce MFA for all user accounts with support for multiple authentication factors.

    "applications": {
      "includeApplications": ["All"]

  "grantControls": {
    "operator": "OR",
    "builtInControls": ["mfa"]
  }
}

# OATH TOTP Implementation (Node.js)
const speakeasy = require('speakeasy');

function generateMFASecret(user) {
    const secret = speakeasy.generateSecret({
        name: `MyApp (${user.email})`,
        issuer: 'MyApp'
    });
    
    // Store secret.base32 securely for user
    return {
        secret: secret.base32,
        qrCode: secret.otpauth_url
    };
}

function verifyMFAToken(userSecret, token) {
    return speakeasy.totp.verify({
        secret: userSecret,
        encoding: 'base32',
        token: token,
        window: 2
    });
}

Validation:

  • Test all MFA methods (TOTP, SMS, push notifications)
  • Verify backup recovery codes work
  • Test MFA bypass scenarios for emergencies
  • Validate MFA enforcement across all applications

# OpenID Connect Implementation (Node.js)
const { Issuer, Strategy } = require('openid-client');

async function setupOIDC() {
    const issuer = await Issuer.discover('https://auth.example.com/.well-known/openid_configuration');
    
    const client = new issuer.Client({
        client_id: 'myapp',
        client_secret: 'secret',
        redirect_uris: ['https://myapp.com/auth/callback'],
        response_types: ['code']
    });
    
    return new Strategy({ client }, (tokenSet, userinfo, done) => {
        return done(null, userinfo);
    });
}

Implement Adaptive Authentication

High Priority High Difficulty

Requirement: Deploy risk-based authentication that adapts security requirements based on context.

Implementation Example:

# Risk Assessment Engine (Python)
import geoip2.database
from datetime import datetime, timedelta

class AdaptiveAuthEngine:
    def __init__(self):
        self.geo_reader = geoip2.database.Reader('GeoLite2-City.mmdb')
        
    def assess_risk(self, user, request_data):
        risk_score = 0
        risk_factors = []
        
        # Geographic risk
        if self.is_unusual_location(user, request_data['ip']):
            risk_score += 30
            risk_factors.append("unusual_location")
            
        # Device risk
        if self.is_new_device(user, request_data['device_fingerprint']):
            risk_score += 20
            risk_factors.append("new_device")
            
        # Time-based risk
        if self.is_unusual_time(user, request_data['timestamp']):
            risk_score += 15
            risk_factors.append("unusual_time")
            
        # Failed attempts
        recent_failures = self.get_recent_failures(user)
        if recent_failures > 3:
            risk_score += 25
            risk_factors.append("multiple_failures")
            
        return {
            'risk_score': risk_score,
            'risk_level': self.get_risk_level(risk_score),
            'factors': risk_factors,
            'required_factors': self.get_required_factors(risk_score)
        }
    
    def get_required_factors(self, risk_score):
        if risk_score >= 50:
            return ['password', 'mfa', 'admin_approval']
        elif risk_score >= 30:
            return ['password', 'mfa']
        else:
            return ['password']

Security Considerations:

  • Balance security with user experience
  • Implement machine learning for better risk assessment
  • Regularly update risk models based on threat intelligence
  • Provide clear feedback to users about security decisions

Implement Role-Based Access Control (RBAC)

High Priority Medium Difficulty

Requirement: Deploy hierarchical RBAC system with inheritance and role separation.

Implementation Example:

# RBAC Database Schema
CREATE TABLE roles (
    id UUID PRIMARY KEY,
    name VARCHAR(100) UNIQUE NOT NULL,
    description TEXT,
    parent_role_id UUID REFERENCES roles(id),
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE permissions (
    id UUID PRIMARY KEY,
    resource VARCHAR(100) NOT NULL,
    action VARCHAR(50) NOT NULL,
    description TEXT,
    UNIQUE(resource, action)
);

CREATE TABLE role_permissions (
    role_id UUID REFERENCES roles(id),
    permission_id UUID REFERENCES permissions(id),
    PRIMARY KEY (role_id, permission_id)
);

CREATE TABLE user_roles (
    user_id UUID REFERENCES users(id),
    role_id UUID REFERENCES roles(id),
    granted_by UUID REFERENCES users(id),
    granted_at TIMESTAMP DEFAULT NOW(),
    expires_at TIMESTAMP,
    PRIMARY KEY (user_id, role_id)
);

# RBAC Implementation (Python)
class RBACEngine:
    def __init__(self, db):
        self.db = db
        
    def check_permission(self, user_id, resource, action):
        query = """
        WITH RECURSIVE role_hierarchy AS (
            SELECT r.id, r.name, r.parent_role_id
            FROM roles r
            JOIN user_roles ur ON r.id = ur.role_id
            WHERE ur.user_id = %s 
              AND (ur.expires_at IS NULL OR ur.expires_at > NOW())
            
            UNION ALL
            
            SELECT r.id, r.name, r.parent_role_id
            FROM roles r
            JOIN role_hierarchy rh ON r.id = rh.parent_role_id
        )
        SELECT COUNT(*) > 0 as has_permission
        FROM role_hierarchy rh
        JOIN role_permissions rp ON rh.id = rp.role_id
        JOIN permissions p ON rp.permission_id = p.id
        WHERE p.resource = %s AND p.action = %s
        """
        
        result = self.db.execute(query, (user_id, resource, action))
        return result.fetchone()['has_permission']
